DROP TABLE IF EXISTS USERS;

CREATE TABLE USERS(
    ID INT AUTO_INCREMENT,
    USER VARCHAR(30) NOT NULL,
    PASSWORD VARCHAR(64) NOT NULL,
    LEVEL INT NOT NULL,
    PRIMARY KEY(ID)
);
